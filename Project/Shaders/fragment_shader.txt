#version 130

in vec2 fragmentPosition;
in vec4 fragmentColor;
out vec4 color;

uniform float time;

void main(){
	color = vec4(fragmentColor.r*(cos(fragmentPosition.x * 4.0 + time)+ 2.0)*0.5,
				fragmentColor.g*(sin(fragmentPosition.y * 3.0 + time)+ 1.0)*0.5,
				fragmentColor.b*(cos(fragmentPosition.x * 2.0 + time)+ 1.0)*0.6,
				fragmentColor.a);

}
